# ============ NEWS FETCHING ============

async def fetch_rss_feed(session: aiohttp.ClientSession, feed_name: str, feed_url: str) -> List[dict]:
    \"\"\"Fetch and parse a single RSS feed\"\"\"
    try:
        async with session.get(feed_url, timeout=aiohttp.ClientTimeout(total=10)) as response:
            if response.status == 200:
                content = await response.text()
                feed = feedparser.parse(content)
                articles = []
                
                for entry in feed.entries[:10]:  # Limit to 10 per feed
                    title = entry.get('title', '')
                    description = entry.get('summary', entry.get('description', ''))
                    
                    # Get image from media content or enclosure
                    image_url = None
                    if 'media_content' in entry:
                        image_url = entry.media_content[0].get('url')
                    elif 'enclosures' in entry and entry.enclosures:
                        image_url = entry.enclosures[0].get('href')
                    elif 'media_thumbnail' in entry:
                        image_url = entry.media_thumbnail[0].get('url')
                    
                    # Parse published date
                    published = entry.get('published_parsed') or entry.get('updated_parsed')
                    if published:
                        published_dt = datetime(*published[:6], tzinfo=timezone.utc)
                        published_str = published_dt.isoformat()
                    else:
                        published_str = datetime.now(timezone.utc).isoformat()
                    
                    articles.append({
                        \"id\": str(uuid.uuid4()),
                        \"title\": title,
                        \"description\": description[:300] if description else None,
                        \"content\": entry.get('content', [{}])[0].get('value') if 'content' in entry else None,
                        \"url\": entry.get('link', ''),
                        \"image_url\": image_url,
                        \"source\": feed_name.upper(),
                        \"author\": entry.get('author'),
                        \"published_at\": published_str,
                        \"platform\": detect_platform(title, description)
                    })
                
                return articles
    except Exception as e:
        logger.error(f\"Error fetching RSS feed {feed_name}: {e}\")
    return []

async def fetch_newsapi_articles(query: str = \"video games\", page_size: int = 20) -> List[dict]:
    \"\"\"Fetch articles from NewsAPI\"\"\"
    if not NEWS_API_KEY:

    try:

                            \"author\": item.get('author'),

@api_router.get(\"/news\", response_model=ArticleResponse)
async def get_news(
    platform: Optional[str] = Query(None, description=\"Filter by platform: pc, playstation, xbox, nintendo, mobile\"),
    search: Optional[str] = Query(None, description=\"Search query\"),
    page: int = Query(1, ge=1),
    page_size: int = Query(20, ge=1, le=100)
):
    \"\"\"Get gaming news with optional filters\"\"\"
    try:
        articles = await fetch_all_news()
        
        # Filter by platform
        if platform and platform != \"all\":
            articles = [a for a in articles if a.get('platform') == platform or a.get('platform') == 'all']
        
        # Filter by search query
        if search:
            search_lower = search.lower()
            articles = [
                a for a in articles 
                if search_lower in a.get('title', '').lower() 
                or search_lower in (a.get('description') or '').lower()
            ]
        
        # Pagination
        total = len(articles)
        start = (page - 1) * page_size
        end = start + page_size
        paginated_articles = articles[start:end]
        
        return ArticleResponse(
            articles=[Article(**a) for a in paginated_articles],
            total=total,
            page=page,
            page_size=page_size
        )
    except Exception as e:
        logger.error(f\"Error fetching news: {e}\")
        raise HTTPException(status_code=500, detail=\"Failed to fetch news\")

async def fetch_rss_feed(session: aiohttp.ClientSession, feed_name: str, feed_url: str) -> List[dict]:
    try:
        async with session.get(feed_url, timeout=aiohttp.ClientTimeout(total=10)) as response:
            if response.status == 200:
                content = await response.text()
                feed = feedparser.parse(content)
                articles = []
                
                for entry in feed.entries[:10]:
                    title = entry.get('title', '')
                    description = entry.get('summary', entry.get('description', ''))
                    
                    image_url = None
                    if 'media_content' in entry:
                        image_url = entry.media_content[0].get('url')
                    elif 'enclosures' in entry and entry.enclosures:
                        image_url = entry.enclosures[0].get('href')
                    elif 'media_thumbnail' in entry:
                        image_url = entry.media_thumbnail[0].get('url')
                    
                    published = entry.get('published_parsed') or entry.get('updated_parsed')
                    if published:
                        published_dt = datetime(*published[:6], tzinfo=timezone.utc)
                        published_str = published_dt.isoformat()
                    else:
                        published_str = datetime.now(timezone.utc).isoformat()
                    
                    articles.append({
                        \"id\": str(uuid.uuid4()),
                        \"title\": title,
                        \"description\": description[:300] if description else None,
                        \"content\": entry.get('content', [{}])[0].get('value') if 'content' in entry else None,
                        \"url\": entry.get('link', ''),
                        \"image_url\": image_url,
                        \"source\": feed_name.upper(),
                        \"author\": entry.get('author'),
                        \"published_at\": published_str,
                        \"platform\": detect_platform(title, description)
                    })
                
                return articles
    except Exception as e:
        logger.error(f\"Error fetching RSS feed {feed_name}: {e}\")
    return []

async def fetch_all_news(enabled_sources: List[str] = None, custom_feeds: List[dict] = None) -> List[dict]:
    all_articles = []
    
    feeds_to_fetch = {}
    
    # Add enabled RSS feeds
    if enabled_sources:
        for source_id in enabled_sources:
            if source_id in RSS_FEEDS:
                feeds_to_fetch[source_id] = RSS_FEEDS[source_id][\"url\"]
    else:
        feeds_to_fetch = {k: v[\"url\"] for k, v in RSS_FEEDS.items()}
    
    # Add custom RSS feeds
    if custom_feeds:
        for feed in custom_feeds:
            feeds_to_fetch[feed.get(\"name\", \"custom\")] = feed.get(\"url\")
    
    # Fetch RSS feeds in parallel
    if feeds_to_fetch:
        async with aiohttp.ClientSession() as session:
            tasks = [fetch_rss_feed(session, name, url) for name, url in feeds_to_fetch.items()]
            results = await asyncio.gather(*tasks, return_exceptions=True)
            
            for result in results:
                if isinstance(result, list):
                    all_articles.extend(result)
    
    # If no articles from live sources, use mock data
    if not all_articles:
        all_articles = MOCK_ARTICLES.copy()
    
    # Sort by published date
    all_articles.sort(key=lambda x: x.get('published_at', ''), reverse=True)
    
    return all_articles

@api_router.get(\"/news\", response_model=ArticleResponse)
async def get_news(
    platform: Optional[str] = Query(None),
    search: Optional[str] = Query(None),
    sources: Optional[str] = Query(None, description=\"Comma-separated source IDs\"),
    page: int = Query(1, ge=1),
    page_size: int = Query(20, ge=1, le=100)
):
    try:
        enabled_sources = sources.split(\",\") if sources else None
        articles = await fetch_all_news(enabled_sources=enabled_sources)
        
        if platform and platform != \"all\":
            articles = [a for a in articles if a.get('platform') == platform or a.get('platform') == 'all']
        
        if search:
            search_lower = search.lower()
            articles = [
                a for a in articles 
                if search_lower in a.get('title', '').lower() 
                or search_lower in (a.get('description') or '').lower()
            ]
        
        total = len(articles)
        start = (page - 1) * page_size
        end = start + page_size
        paginated_articles = articles[start:end]
        
        return ArticleResponse(
            articles=[Article(**a) for a in paginated_articles],
            total=total,
            page=page,
            page_size=page_size
        )
    except Exception as e:
        logger.error(f\"Error fetching news: {e}\")
        raise HTTPException(status_code=500, detail=\"Failed to fetch news\")

const formatRelativeTime = (dateString) => {
  const date = new Date(dateString);
  const now = new Date();
  const diffMs = now - date;
  const diffMins = Math.floor(diffMs / 60000);
  const diffHours = Math.floor(diffMs / 3600000);
  const diffDays = Math.floor(diffMs / 86400000);
  if (diffMins < 60) return `Il y a ${diffMins} min`;
  if (diffHours < 24) return `Il y a ${diffHours}h`;
  if (diffDays < 7) return `Il y a ${diffDays}j`;
  return date.toLocaleDateString('fr-FR');
};

const applyTheme = (themeId) => {
  const theme = THEMES[themeId];
  if (!theme) return;
  const root = document.documentElement;
  Object.entries(theme.colors).forEach(([key, value]) => {
    root.style.setProperty(`--${key}`, value);
  });
};

                  <Button variant={settings?.display_mode === 'list' ? 'default' : 'outline'} size=\"sm\" onClick={() => onUpdateSettings({ display_mode: 'list' })} data-testid=\"display-list-btn\"><List size={16} /></Button>
                </div>
              </div>

              <div className=\"settings-option\">
                <div>
                  <div className=\"settings-option-label\">Taille des cartes</div>
                </div>
                <div className=\"flex gap-2\">
                  {['compact', 'normal', 'large'].map((size) => (
                    <Button key={size} variant={settings?.card_size === size ? 'default' : 'outline'} size=\"sm\" onClick={() => onUpdateSettings({ card_size: size })} data-testid={`card-size-${size}`}>{size.charAt(0).toUpperCase() + size.slice(1)}</Button>
                  ))}

              <div className=\"settings-option\">
                <div>
                  <div className=\"settings-option-label\">Densité</div>
                </div>
                <div className=\"flex gap-2\">
                  {['compact', 'comfortable', 'spacious'].map((d) => (
                    <Button key={d} variant={settings?.density === d ? 'default' : 'outline'} size=\"sm\" onClick={() => onUpdateSettings({ density: d })} data-testid={`density-${d}`}>{d.charAt(0).toUpperCase() + d.slice(1)}</Button>
                  ))}
                </div>
              </div>

              <div className=\"settings-option\">
                <div>
                  <div className=\"settings-option-label\">Animations</div>
                  <div className=\"settings-option-description\">Activer les animations</div>
                </div>
                <Switch checked={settings?.animations_enabled !== false} onCheckedChange={(v) => onUpdateSettings({ animations_enabled: v })} data-testid=\"animations-toggle\" />
              </div>
            </div>

            {/* SOURCES */}
            <div className=\"settings-section\">
              <div className=\"settings-section-title\"><Rss size={18} /> Sources d'actualités</div>
              
              <div className=\"space-y-2 mb-4\">
                {Object.entries(RSS_SOURCES).map(([id, source]) => (
                  <div key={id} className=\"flex items-center justify-between p-2 rounded bg-[var(--surface)]\">
                    <span>{source.logo} {source.name}</span>
                    <Switch checked={!settings?.enabled_sources || settings.enabled_sources.length === 0 || settings.enabled_sources.includes(id)} onCheckedChange={(v) => {
                      const current = settings?.enabled_sources || Object.keys(RSS_SOURCES);
                      const updated = v ? [...current, id] : current.filter(s => s !== id);
                      onUpdateSettings({ enabled_sources: updated });
                    }} data-testid={`source-toggle-${id}`} />
                  </div>
                ))}
              </div>

              <div className=\"p-3 border border-[var(--border)] rounded-lg\">
                <div className=\"text-sm font-semibold mb-2\">Ajouter une source RSS</div>
                <Input placeholder=\"Nom\" value={newSourceName} onChange={(e) => setNewSourceName(e.target.value)} className=\"mb-2\" data-testid=\"custom-source-name\" />
                <Input placeholder=\"URL du flux RSS\" value={newSourceUrl} onChange={(e) => setNewSourceUrl(e.target.value)} className=\"mb-2\" data-testid=\"custom-source-url\" />
                <Button onClick={handleAddSource} className=\"w-full\" data-testid=\"add-custom-source-btn\"><Plus size={16} className=\"mr-2\" /> Ajouter</Button>
              </div>

              {settings?.custom_rss_feeds?.length > 0 && (
                <div className=\"mt-4\">
                  <div className=\"text-sm font-semibold mb-2\">Sources personnalisées</div>

                      <span>📡 {feed.name}</span>
                      <Button variant=\"ghost\" size=\"sm\" onClick={() => onRemoveCustomSource(feed.name)} data-testid={`remove-custom-source-${idx}`}><Trash2 size={16} /></Button>

                  ))}

              )}
            </div>

            {/* SOUNDS */}
            <div className=\"settings-section\">
              <div className=\"settings-section-title\"><Volume2 size={18} /> Sons & Notifications</div>
              
              <div className=\"settings-option\">
                <div>
                  <div className=\"settings-option-label\">Sons activés</div>
                </div>
                <Switch checked={settings?.sound_enabled !== false} onCheckedChange={(v) => onUpdateSettings({ sound_enabled: v })} data-testid=\"sounds-toggle\" />
              </div>

              {settings?.sound_enabled !== false && (
                <>
                  <div className=\"settings-option\">

                    </div>
                  </div>

                  <div className=\"settings-option flex-col items-start gap-2\">

                        <Button key={id} variant={settings?.notification_sound === id ? 'default' : 'outline'} size=\"sm\" onClick={() => onUpdateSettings({ notification_sound: id })} data-testid={`sound-${id}`}>{sound.name}</Button>

              )}

            {/* KEYWORD ALERTS */}
            <div className=\"settings-section\">

              <div className=\"settings-section-title\"><RefreshCw size={18} /> Actualisation</div>

                </div>

                </div>
              )}
            </div>

          </div>
        </ScrollArea>
      </div>
    </>

};

// ============ MAIN APP ============

function App() {
  const [articles, setArticles] = useState([]);
  const [favorites, setFavorites] = useState([]);
  const [trendingArticles, setTrendingArticles] = useState([]);
  const [isLoading, setIsLoading] = useState(true);
  const [currentPlatform, setCurrentPlatform] = useState('all');
  const [searchQuery, setSearchQuery] = useState('');
  const [showFavorites, setShowFavorites] = useState(false);
  const [sidebarOpen, setSidebarOpen] = useState(false);
  const [settingsOpen, setSettingsOpen] = useState(false);
  const [settings, setSettings] = useState({
    theme: 'cyberpunk',
    display_mode: 'grid',
    card_size: 'normal',
    density: 'comfortable',
    animations_enabled: true,
    sound_enabled: true,
    sound_volume: 50,
    notification_sound: 'level_up',
    music_volume: 30,
    auto_refresh: true,
    refresh_interval: 300,
    enabled_sources: [],
    custom_rss_feeds: [],
    keyword_alerts: []
  });
  const [keywordAlerts, setKeywordAlerts] = useState([]);

  // Fetch settings
  const fetchSettings = useCallback(async () => {
    try {
      const response = await axios.get(`${API}/settings`);
      setSettings(response.data);
      setKeywordAlerts(response.data.keyword_alerts || []);
      if (response.data.theme) applyTheme(response.data.theme);
    } catch (error) {
      console.error('Error fetching settings:', error);
    }
  }, []);

  // Update settings
  const updateSettings = async (updates) => {
    try {
      const response = await axios.put(`${API}/settings`, updates);
      setSettings(response.data);
      if (updates.theme) applyTheme(updates.theme);
      toast.success('Paramètres sauvegardés');
    } catch (error) {
      console.error('Error updating settings:', error);
      toast.error('Erreur lors de la sauvegarde');
    }
  };

  // Fetch news
  const fetchNews = useCallback(async () => {
    setIsLoading(true);
    try {
      const params = new URLSearchParams();
      if (currentPlatform && currentPlatform !== 'all') params.append('platform', currentPlatform);
      if (searchQuery) params.append('search', searchQuery);
      if (settings.enabled_sources?.length) params.append('sources', settings.enabled_sources.join(','));
      params.append('page_size', '40');

      const response = await axios.get(`${API}/news?${params.toString()}`);
      setArticles(response.data.articles || []);

      // Check keyword alerts
      if (keywordAlerts.length > 0 && settings.sound_enabled) {
        const newArticles = response.data.articles || [];
        newArticles.forEach(article => {
          keywordAlerts.forEach(keyword => {
            if (article.title.toLowerCase().includes(keyword.toLowerCase())) {
              toast(`🔔 Alerte: \"${keyword}\" - ${article.title}`);
            }
          });
        });
      }
    } catch (error) {
      console.error('Error fetching news:', error);
      toast.error('Erreur lors du chargement des actualités');
    } finally {
      setIsLoading(false);
    }
  }, [currentPlatform, searchQuery, settings.enabled_sources, keywordAlerts, settings.sound_enabled]);

  // Fetch trending
  const fetchTrending = useCallback(async () => {
    try {
      const response = await axios.get(`${API}/news/trending?page_size=10`);
      setTrendingArticles(response.data.articles || []);
    } catch (error) {
      console.error('Error fetching trending:', error);
    }
  }, []);

  // Fetch favorites
  const fetchFavorites = useCallback(async () => {
    try {
      const response = await axios.get(`${API}/favorites`);
      setFavorites(response.data || []);
    } catch (error) {
      console.error('Error fetching favorites:', error);
    }
  }, []);

  // Toggle favorite
  const handleToggleFavorite = async (article) => {
    const isFavorite = favorites.some(f => f.article_id === article.id);
    try {
      if (isFavorite) {
        await axios.delete(`${API}/favorites/${article.id}`);
        setFavorites(prev => prev.filter(f => f.article_id !== article.id));
        toast.success('Retiré des favoris');
      } else {
        const response = await axios.post(`${API}/favorites`, { article_id: article.id, article_data: article });
        setFavorites(prev => [...prev, response.data]);
        toast.success('Ajouté aux favoris');
      }
    } catch (error) {
      console.error('Error toggling favorite:', error);
      toast.error('Erreur');
    }
  };

  // Add custom source
  const addCustomSource = async (name, url) => {
    try {
      await axios.post(`${API}/sources/custom`, { name, url });
      await fetchSettings();
      toast.success('Source ajoutée');
    } catch (error) {
      console.error('Error adding custom source:', error);
      toast.error('Erreur: URL invalide ou inaccessible');
    }
  };

  // Remove custom source
  const removeCustomSource = async (name) => {
    try {
      await axios.delete(`${API}/sources/custom/${name}`);
      await fetchSettings();
      toast.success('Source supprimée');
    } catch (error) {
      console.error('Error removing custom source:', error);
    }
  };

  // Add keyword alert
  const addKeywordAlert = async (keyword) => {
    try {
      await axios.post(`${API}/alerts/keyword?keyword=${encodeURIComponent(keyword)}`);
      setKeywordAlerts(prev => [...prev, keyword]);
      toast.success(`Alerte ajoutée pour \"${keyword}\"`);
    } catch (error) {
      console.error('Error adding keyword alert:', error);
    }
  };

  // Remove keyword alert
  const removeKeywordAlert = async (keyword) => {
    try {
      await axios.delete(`${API}/alerts/keyword/${encodeURIComponent(keyword)}`);
      setKeywordAlerts(prev => prev.filter(k => k !== keyword));
      toast.success(`Alerte supprimée`);
    } catch (error) {
      console.error('Error removing keyword alert:', error);
    }
  };

  // Debounced search
  useEffect(() => {
    const timer = setTimeout(() => { if (!showFavorites) fetchNews(); }, 300);
    return () => clearTimeout(timer);
  }, [searchQuery, currentPlatform, showFavorites, fetchNews]);

  // Initial load
  useEffect(() => {
    fetchSettings();
    fetchNews();
    fetchTrending();
    fetchFavorites();
  }, []);

  // Auto refresh
  useEffect(() => {
    if (settings.auto_refresh && settings.refresh_interval > 0) {
      const interval = setInterval(() => {
        fetchNews();
        fetchTrending();
      }, settings.refresh_interval * 1000);
      return () => clearInterval(interval);
    }
  }, [settings.auto_refresh, settings.refresh_interval, fetchNews, fetchTrending]);

  // Apply theme on load
  useEffect(() => {
    if (settings.theme) applyTheme(settings.theme);
  }, [settings.theme]);

  return (
    <SettingsContext.Provider value={settings}>
      <div className=\"app-container\" data-testid=\"app-container\">
        <Toaster position=\"bottom-right\" richColors theme=\"dark\" />
        
        <Sidebar
          currentPlatform={currentPlatform}
          onPlatformChange={setCurrentPlatform}
          favoritesCount={favorites.length}
          showFavorites={showFavorites}
          onToggleFavorites={setShowFavorites}
          isOpen={sidebarOpen}
          onClose={() => setSidebarOpen(false)}
          onOpenSettings={() => setSettingsOpen(true)}
        />

        <main className=\"main-content\" style={{ paddingBottom: '80px' }}>
          <Header
            searchQuery={searchQuery}
            onSearchChange={setSearchQuery}
            onMenuToggle={() => setSidebarOpen(!sidebarOpen)}
            onOpenSettings={() => setSettingsOpen(true)}
          />

          <TrendingTicker articles={trendingArticles} settings={settings} />

          <NewsFeed
            articles={articles}
            favorites={favorites}
            onToggleFavorite={handleToggleFavorite}
            isLoading={isLoading}
            showFavorites={showFavorites}
            settings={settings}
          />
        </main>

        <MusicPlayer settings={settings} onUpdateSettings={updateSettings} />

        <SettingsPanel
          isOpen={settingsOpen}
          onClose={() => setSettingsOpen(false)}
          settings={settings}
          onUpdateSettings={updateSettings}
          sources={RSS_SOURCES}
          onAddCustomSource={addCustomSource}
          onRemoveCustomSource={removeCustomSource}
          keywordAlerts={keywordAlerts}
          onAddKeywordAlert={addKeywordAlert}
          onRemoveKeywordAlert={removeKeywordAlert}
        />
      </div>
    </SettingsContext.Provider>
  );
}

export default App;
